{{- /* Schema.org VideoObject structured data for pages with YouTube embeds */ -}} {{- $content := .RawContent -}} {{- $youtubeRegex := `\{\{<\s*youtube\s+([a-zA-Z0-9_-]+)\s*>}}` -}} {{- $match := findRE $youtubeRegex $content 1 -}} {{- if $match -}} {{- $fullMatch := index $match 0 -}} {{- $videoId := replaceRE `\{\{<\s*youtube\s+` "" $fullMatch -}} {{- $videoId = replaceRE `\s*>}}` "" $videoId -}} {{- $videoId = trim $videoId " " -}} {{- if $videoId -}} {{- $desc := or .Description .Summary | plainify | truncate 200 -}} {{- end -}} {{- end -}}
